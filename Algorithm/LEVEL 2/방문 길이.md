def solution(dirs):
    command={'U':(0,1),'D':(0,-1),'R':(1,0),'L':(-1,0)}
    cur_x=0
    cur_y=0
    move=[]
    
    for pos in dirs:
        nex_x,nex_y=cur_x+command[pos][0],cur_y+command[pos][1]
        if -5<=nex_x<=5 and -5<=nex_y<=5:
            move.append((cur_x,cur_y,nex_x,nex_y))
            move.append((nex_x,nex_y,cur_x,cur_y))
            cur_x=nex_x
            cur_y=nex_y
            
    move=set(move)
        
    return len(move)/2
